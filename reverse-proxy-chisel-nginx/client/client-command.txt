# Запускать на локальном ПК.
# Локальные порты НЕ меняются: приложение на 127.0.0.1:8050, WebSocket на 127.0.0.1:8765.
# Замените SERVER_PUBLIC_IP, CHISEL_USER и CHISEL_PASSWORD.

chisel client \
  --auth CHISEL_USER:CHISEL_PASSWORD \
  SERVER_PUBLIC_IP:4016 \
  R:0.0.0.0:18050:127.0.0.1:8050 \
  R:0.0.0.0:4014:127.0.0.1:8765
